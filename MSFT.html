import datetime

def str_to_datetime(s):
    split = s.split('-')   # Splitting the date string by hyphens to extract year, month, and day
    year,month,day = int(split[0]),int(split[1]),int(split[2])  # Converting split parts into integers
    return datetime.datetime(year=year, month=month, day=day)   # Creating a datetime object

import numpy as np

# Convert date strings to datetime objects
def df_to_windowed_df(dataframe, first_date_str, last_date_str, n=3):
  first_date = str_to_datetime(first_date_str)
  last_date  = str_to_datetime(last_date_str)
  
  # Initialize variables to store dates, input features (X), and output values (Y)
  target_date = first_date
  
  dates = []
  X, Y = [], []

  last_time = False
  while True:
    df_subset = dataframe.loc[:target_date].tail(n+1)  ## Extract a subset of the DataFrame up to the target date
    
    # Check if the subset size matches the window size
    if len(df_subset) != n+1:
      print(f'Error: Window of size {n} is too large for date {target_date}') 
      return

    # Extract input and output values from the subse
    values = df_subset['Close'].to_numpy()
    x, y = values[:-1], values[-1]

    # Store dates, input features, and output values
    dates.append(target_date)
    X.append(x)
    Y.append(y)

    # Determine the next week's date for iteration
    next_week = dataframe.loc[target_date:target_date+datetime.timedelta(days=7)]
    next_datetime_str = str(next_week.head(2).tail(1).index.values[0])
    next_date_str = next_datetime_str.split('T')[0]
    year_month_day = next_date_str.split('-')
    year, month, day = year_month_day
    next_date = datetime.datetime(day=int(day), month=int(month), year=int(year))
    
    # Check if the last date has been reached
    if last_time:
      break
    
    target_date = next_date

    if target_date == last_date:
      last_time = True

  # Create a new DataFrame to organize the collected data   
  ret_df = pd.DataFrame({})
  ret_df['Target Date'] = dates
  
  X = np.array(X)
  for i in range(0, n):
    X[:, i]
    ret_df[f'Target-{n-i}'] = X[:, i]
  
  # Store output values in the new DataFrame
  ret_df['Target'] = Y

  return ret_df

def windowed_df_to_date_X_y(windowed_dataframe):
  # Convert windowed DataFrame to a NumPy array
  df_as_np = windowed_dataframe.to_numpy()
  
  # Extract dates, input features (X), and output values (Y) from the array
  dates = df_as_np[:, 0] 

  # Extract input features, reshape into a 3D array for model input
  middle_matrix = df_as_np[:, 1:-1]
  X = middle_matrix.reshape((len(dates), middle_matrix.shape[1], 1))

  # Extract output values
  Y = df_as_np[:, -1]

  return dates, X.astype(np.float32), Y.astype(np.float32)

  1. Prediction Errors: Differences between the lines could also indicate areas where the model struggles to make accurate predictions, pointing to instances where the model requires improvement or where the data might be inherently noisy or complex.
